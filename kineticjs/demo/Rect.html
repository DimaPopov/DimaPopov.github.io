 var stage = new Kinetic.Stage({
      container: 'kinetic-container',
      width: 600,
      height: 400
  });
  var layer = new Kinetic.Layer();
  var rect = new Kinetic.Rect({
      x: 100,
      y: 150,
      width: 75,
      height: 225,
      fill: 'yellowgreen',
      stroke: 'purple',
      lineWidth: 5,
      draggable: true
  });
  stage.add(layer);
  layer.add(rect);
  stage.draw();
