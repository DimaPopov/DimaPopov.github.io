  var width = 600;
  var height = 400;

  function tango(layer) {
      for (var n = 0; n < layer.getChildren().length; n++) {
          var shape = layer.getChildren()[n];
          var stage = shape.getStage();
          var radius = Math.random() * 100 + 20;
          new Kinetic.Tween({
              node: shape,
              duration: 1,
              x: Math.random() * stage.width(),
              y: Math.random() * stage.height(),
              rotation: Math.random() * 360,
              radius: radius,
              opacity: (radius - 20) / 100,
              easing: Kinetic.Easings.EaseInOut,
              fillRed: 355 / (Math.random() * 10),
              fillGreen: 255 / (Math.random() * 10),
              fillBlue: 255 / (Math.random() * 10)
          }).play();
      }
  }
  var stage = new Kinetic.Stage({
      container: 'kinetic-container',
      width: width,
      height: height
  });
  var layer = new Kinetic.Layer();
  for (var n = 0; n < 10; n++) {
      var radius = Math.random() * 100 + 20;
      var shape = new Kinetic.RegularPolygon({
          x: Math.random() * stage.getWidth(),
          y: Math.random() * stage.getHeight(),
          sides: Math.ceil(Math.random() * 5 + 3),
          radius: radius,
          fillRed: 355 / (Math.random() * 10),
          fillGreen: 255 / (Math.random() * 10),
          fillBlue: 255 / (Math.random() * 10),
          opacity: (radius - 20) / 100,
          draggable: true
      });
      layer.add(shape);
  }
  stage.add(layer);
  document.getElementById('tango').addEventListener('click', function() {
      tango(layer);
  }, false);
