Kinetic.Label





 var stage = new Kinetic.Stage({
     container: 'kinetic-container',
     width: 600,
     height: 400
 });
 var layer = new Kinetic.Layer();
 // create label
 var label = new Kinetic.Label({
     x: 100,
     y: 100,
     draggable: true
 });
 // add a tag to the label
 label.add(new Kinetic.Tag({
     fill: '#bbb',
     stroke: '#333',
     shadowColor: 'black',
     shadowBlur: 10,
     shadowOffset: [10, 10],
     shadowOpacity: 0.2,
     lineJoin: 'round',
     pointerDirection: 'up',
     pointerWidth: 20,
     pointerHeight: 20,
     cornerRadius: 5
 }));
 // add text to the label
 label.add(new Kinetic.Text({
     text: 'Hello World!',
     fontSize: 50,
     lineHeight: 1.2,
     padding: 10,
     fill: 'green'
 }));
 stage.add(layer);
 layer.add(label);
 stage.draw();
