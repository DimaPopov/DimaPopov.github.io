Kinetic.Sprite






var width = 600;
var height = 400;
var stage = new Kinetic.Stage({
    container: 'kinetic-container',
    width: width,
    height: height
});
var layer = new Kinetic.Layer();
var animations = {
    idle: [
        10, 0, 170, 150,
        190, 0, 170, 150,
        375, 0, 170, 150,
        550, 0, 175, 150,
        725, 0, 200, 150,
        550, 0, 175, 150,
        375, 0, 170, 150,
        190, 0, 170, 150
    ]
};
var imageObj = new Image();
imageObj.onload = function() {
    var spite = new Kinetic.Sprite({
        x: 50,
        y: 50,
        image: imageObj,
        animation: 'idle',
        animations: animations,
        frameRate: 10,
        frameIndex: 0
    });
    // add the shape to the layer
    layer.add(spite);
    // add the layer to the stage
    stage.add(layer);
    // start sprite animation
    spite.start();
};
imageObj.src = '../image/bird-sprite.png';
