Kinetic.Ellipse





var stage = new Kinetic.Stage({
    container: 'kinetic-container',
    width: 600,
    height: 400
});
var layer = new Kinetic.Layer();
var ellipse = new Kinetic.Ellipse({
    x: 100,
    y: 200,
    radius: {
        x: 80,
        y: 50,
    },
    fill: 'red',
    draggable: true
});
stage.add(layer);
layer.add(ellipse);
stage.draw();
