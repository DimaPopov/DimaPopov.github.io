Kinetic.Star





 var stage = new Kinetic.Stage({
     container: 'kinetic-container',
     width: 600,
     height: 400
 });
 var layer = new Kinetic.Layer();
 var star = new Kinetic.Star({
     x: 100,
     y: 200,
     numPoints: 5,
     innerRadius: 30,
     outerRadius: 70,
     fill: 'red',
     stroke: 'black',
     strokeWidth: 4,
     draggable: true
 });
 stage.add(layer);
 layer.add(star);
 stage.draw();
