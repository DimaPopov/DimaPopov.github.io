Kinetic.Wedge





var stage = new Kinetic.Stage({
    container: 'kinetic-container',
    width: 600,
    height: 400
});
var layer = new Kinetic.Layer();
var wedge = new Kinetic.Wedge({
    x: 100,
    y: 200,
    radius: 100,
    fill: 'red',
    stroke: 'black',
    strokeWidth: 5,
    angleDeg: 80,
    rotationDeg: -120,
    draggable: true
});
wedge.offsetX(40);
wedge.dragDistance(2);
stage.add(layer);
layer.add(wedge);
stage.draw();
