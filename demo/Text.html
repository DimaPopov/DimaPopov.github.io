    var stage = new Kinetic.Stage({
        container: 'kinetic-container',
        width: 600,
        height: 400
    });
    var layer = new Kinetic.Layer();
    var text = new Kinetic.Text({
        x: 10,
        y: 15,
        text: 'Simple Text',
        fontSize: 30,
        fontFamily: 'Calibri',
        fill: 'green',
        draggable: true
    });
    stage.add(layer);
    layer.add(text);
    stage.draw();
