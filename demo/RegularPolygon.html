Kinetic.RegularPolygon





   var stage = new Kinetic.Stage({
       container: 'kinetic-container',
       width: 600,
       height: 400
   });
   var layer = new Kinetic.Layer();
   var hexagon1 = new Kinetic.RegularPolygon({
       x: 100,
       y: 200,
       sides: 4,
       radius: 70,
       fill: 'red',
       stroke: 'black',
       strokeWidth: 4,
       draggable: true
   });
   var hexagon2 = new Kinetic.RegularPolygon({
       x: 100,
       y: 200,
       sides: 6,
       radius: 70,
       fill: 'green',
       stroke: 'black',
       strokeWidth: 4,
       draggable: true
   });
   var hexagon3 = new Kinetic.RegularPolygon({
       x: 100,
       y: 200,
       sides: 3,
       radius: 70,
       fill: 'blue',
       stroke: 'black',
       strokeWidth: 4,
       draggable: true
   });
   stage.add(layer);
   layer.add(hexagon1);
   layer.add(hexagon2);
   layer.add(hexagon3);
   stage.draw();
